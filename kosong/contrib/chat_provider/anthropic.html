145    async def generate(
146        self,
147        system_prompt: str,
148        tools: Sequence[Tool],
149        history: Sequence[Message],
150    ) -> "AnthropicStreamedMessage":
151        # https://docs.claude.com/en/api/messages#body-messages
152        # Anthropic API does not support system roles, but just a system prompt.
153        system = (
154            [
155                TextBlockParam(
156                    text=system_prompt,
157                    type="text",
158                    cache_control=CacheControlEphemeralParam(type="ephemeral"),
159                )
160            ]
161            if system_prompt
162            else omit
163        )
164        messages: list[MessageParam] = []
165        for message in history:
166            messages.append(self._convert_message(message))
167        if messages:
168            last_message = messages[-1]
169            last_content = last_message["content"]
170
171            # inject cache control in the last content.
172            # https://docs.claude.com/en/docs/build-with-claude/prompt-caching
173            if isinstance(last_content, list) and last_content:
174                content_blocks = cast(list[ContentBlockParam], last_content)
175                last_block = content_blocks[-1]
176                match last_block["type"]:
177                    case (
178                        "text"
179                        | "image"
180                        | "document"
181                        | "search_result"
182                        | "tool_use"
183                        | "tool_result"
184                        | "server_tool_use"
185                        | "web_search_tool_result"
186                    ):
187                        last_block["cache_control"] = CacheControlEphemeralParam(type="ephemeral")
188                    case "thinking" | "redacted_thinking":
189                        pass
190        generation_kwargs: dict[str, Any] = {}
191        generation_kwargs.update(self._generation_kwargs)
192        betas = generation_kwargs.pop("beta_features", [])
193        extra_headers = {
194            **{"anthropic-beta": ",".join(str(e) for e in betas)},
195            **(generation_kwargs.pop("extra_headers", {})),
196        }
197
198        tools_ = [_convert_tool(tool) for tool in tools]
199        if tools:
200            tools_[-1]["cache_control"] = CacheControlEphemeralParam(type="ephemeral")
201        try:
202            response = await self._client.messages.create(
203                model=self._model,
204                messages=messages,
205                system=system,
206                tools=tools_,
207                stream=self._stream,
208                extra_headers=extra_headers,
209                **generation_kwargs,
210            )
211            return AnthropicStreamedMessage(response)
212        except AnthropicError as e:
213            raise _convert_error(e) from e

253    def _convert_message(self, message: Message) -> MessageParam:
254        """Convert a single internal message into Anthropic wire format."""
255        role = message.role
256
257        if role == "system":
258            # Anthropic does not support system messages in the conversation.
259            # We map it to a special user message.
260            return MessageParam(
261                role="user",
262                content=[
263                    TextBlockParam(
264                        type="text", text=f"<system>{message.extract_text(sep='\n')}</system>"
265                    )
266                ],
267            )
268        elif role == "tool":
269            if message.tool_call_id is None:
270                raise ChatProviderError("Tool message missing `tool_call_id`.")
271            if self._tool_message_conversion == "extract_text":
272                content = message.extract_text(sep="\n")
273            else:
274                content = message.content
275            block = _tool_result_message_to_block(message.tool_call_id, content)
276            return MessageParam(role="user", content=[block])
277
278        assert role in ("user", "assistant")
279        blocks: list[ContentBlockParam] = []
280        for part in message.content:
281            if isinstance(part, TextPart):
282                blocks.append(TextBlockParam(type="text", text=part.text))
283            elif isinstance(part, ImageURLPart):
284                blocks.append(_image_url_part_to_anthropic(part))
285            elif isinstance(part, ThinkPart):
286                if part.encrypted is None:
287                    # missing signature, strip this thinking block.
288                    continue
289                else:
290                    blocks.append(
291                        ThinkingBlockParam(
292                            type="thinking", thinking=part.think, signature=part.encrypted
293                        )
294                    )
295            else:
296                continue
297        for tool_call in message.tool_calls or []:
298            if tool_call.function.arguments:
299                try:
300                    parsed_arguments = json.loads(tool_call.function.arguments)
301                except json.JSONDecodeError as exc:  # pragma: no cover - defensive guard
302                    raise ChatProviderError("Tool call arguments must be valid JSON.") from exc
303                if not isinstance(parsed_arguments, dict):
304                    raise ChatProviderError("Tool call arguments must be a JSON object.")
305                tool_input = cast(dict[str, object], parsed_arguments)
306            else:
307                tool_input = {}
308            blocks.append(
309                ToolUseBlockParam(
310                    type="tool_use",
311                    id=tool_call.id,
312                    name=tool_call.function.name,
313                    input=tool_input,
314                )
315            )
316        return MessageParam(role=role, content=blocks)

320    def __init__(self, response: AnthropicMessage | AsyncStream[RawMessageStreamEvent]):
321        if isinstance(response, AnthropicMessage):
322            self._iter = self._convert_non_stream_response(response)
323        else:
324            self._iter = self._convert_stream_response(response)
325        self._id: str | None = None
326        self._usage = Usage(input_tokens=0, output_tokens=0)

328    def __aiter__(self) -> AsyncIterator[StreamedMessagePart]:
329        return self

331    async def __anext__(self) -> StreamedMessagePart:
332        return await self._iter.__anext__()

334    @property
335    def id(self) -> str | None:
336        return self._id

338    @property
339    def usage(self) -> TokenUsage | None:
340        # https://docs.claude.com/en/docs/build-with-claude/prompt-caching#tracking-cache-performance
341        return TokenUsage(
342            # Note: in some Anthropic-compatible APIs, input_tokens can be None
343            input_other=self._usage.input_tokens or 0,
344            output=self._usage.output_tokens,
345            input_cache_read=self._usage.cache_read_input_tokens or 0,
346            input_cache_creation=self._usage.cache_creation_input_tokens or 0,
347        )

349    def _update_usage(self, delta_usage: MessageDeltaUsage) -> None:
350        if delta_usage.cache_creation_input_tokens is not None:
351            self._usage.cache_creation_input_tokens = delta_usage.cache_creation_input_tokens
352        if delta_usage.cache_read_input_tokens is not None:
353            self._usage.cache_read_input_tokens = delta_usage.cache_read_input_tokens
354        if delta_usage.input_tokens is not None:
355            self._usage.input_tokens = delta_usage.input_tokens
356        if delta_usage.output_tokens is not None:  # type: ignore
357            self._usage.output_tokens = delta_usage.output_tokens

359    async def _convert_non_stream_response(
360        self,
361        response: AnthropicMessage,
362    ) -> AsyncIterator[StreamedMessagePart]:
363        self._id = response.id
364        self._usage = response.usage
365        for block in response.content:
366            match block.type:
367                case "text":
368                    yield TextPart(text=block.text)
369                case "thinking":
370                    yield ThinkPart(think=block.thinking, encrypted=block.signature)
371                case "redacted_thinking":
372                    yield ThinkPart(think="", encrypted=block.data)
373                case "tool_use":
374                    yield ToolCall(
375                        id=block.id,
376                        function=ToolCall.FunctionBody(
377                            name=block.name, arguments=json.dumps(block.input)
378                        ),
379                    )
380                case _:
381                    continue
382
383    async def _convert_stream_response(
384        self,
385        manager: AsyncStream[RawMessageStreamEvent],
386    ) -> AsyncIterator[StreamedMessagePart]:
387        try:
388            async with manager as stream:
389                async for event in stream:
390                    if isinstance(event, MessageStartEvent):
391                        self._id = event.message.id
392                        # Capture initial usage from start event
393                        # (contains initial prompt/input token usage)
394                        self._usage = event.message.usage
395                    elif isinstance(event, RawContentBlockStartEvent):
396                        block = event.content_block
397                        match block.type:
398                            case "text":
399                                yield TextPart(text=block.text)
400                            case "thinking":
401                                yield ThinkPart(think=block.thinking)
402                            case "redacted_thinking":
403                                yield ThinkPart(think="", encrypted=block.data)
404                            case "tool_use":
405                                yield ToolCall(
406                                    id=block.id,
407                                    function=ToolCall.FunctionBody(name=block.name, arguments=""),
408                                )
409                            case "server_tool_use" | "web_search_tool_result":
410                                # ignore
411                                continue
412                    elif isinstance(event, RawContentBlockDeltaEvent):
413                        delta = event.delta
414                        match delta.type:
415                            case "text_delta":
416                                yield TextPart(text=delta.text)
417                            case "thinking_delta":
418                                yield ThinkPart(think=delta.thinking)
419                            case "input_json_delta":
420                                yield ToolCallPart(arguments_part=delta.partial_json)
421                            case "signature_delta":
422                                yield ThinkPart(think="", encrypted=delta.signature)
423                            case "citations_delta":
424                                # ignore
425                                continue
426                    elif isinstance(event, MessageDeltaEvent):
427                        if event.usage:
428                            self._update_usage(event.usage)
429                    elif isinstance(event, MessageStopEvent):
430                        continue
431        except AnthropicError as exc:
432            raise _convert_error(exc) from exc

443def _tool_result_message_to_block(
444    tool_call_id: str, content: str | list[ContentPart]
445) -> ToolResultBlockParam:
446    block_content: str | list[ToolResultContent]
447    # If tool_result_process is `extract_text`, we join all text parts into one string
448    if isinstance(content, str):
449        block_content = content
450    else:
451        # Otherwise, map parts to content blocks
452        blocks: list[ToolResultContent] = []
453        for part in content:
454            if isinstance(part, TextPart):
455                if part.text:
456                    blocks.append(TextBlockParam(type="text", text=part.text))
457            elif isinstance(part, ImageURLPart):
458                blocks.append(_image_url_part_to_anthropic(part))
459            else:
460                # https://docs.claude.com/en/docs/build-with-claude/files#file-types-and-content-blocks
461                # Anthropic API supports very limited file types
462                raise ChatProviderError(
463                    f"Anthropic API does not support {type(part)} in tool result"
464                )
465        block_content = blocks
466
467    return ToolResultBlockParam(
468        type="tool_result",
469        tool_use_id=tool_call_id,
470        content=block_content,
471    )

474def _image_url_part_to_anthropic(part: ImageURLPart) -> ImageBlockParam:
475    url = part.image_url.url
476    # data:[<media-type>][;base64],<data>
477    if url.startswith("data:"):
478        res = url[5:].split(";base64,", 1)
479        if len(res) != 2:
480            raise ChatProviderError(f"Invalid data URL for image: {url}")
481        media_type, data = res
482        if media_type not in ("image/png", "image/jpeg", "image/gif", "image/webp"):
483            raise ChatProviderError(
484                f"Unsupported media type for base64 image: {media_type}, url: {url}"
485            )
486        return ImageBlockParam(
487            type="image",
488            source=Base64ImageSourceParam(
489                type="base64",
490                data=data,
491                media_type=media_type,
492            ),
493        )
494    else:
495        return ImageBlockParam(
496            type="image",
497            source=URLImageSourceParam(type="url", url=url),
498        )

501def _convert_error(error: AnthropicError) -> ChatProviderError:
502    if isinstance(error, AnthropicAPIStatusError):
503        return APIStatusError(error.status_code, str(error))
504    if isinstance(error, AnthropicAuthenticationError):
505        return APIStatusError(getattr(error, "status_code", 401), str(error))
506    if isinstance(error, AnthropicPermissionDeniedError):
507        return APIStatusError(getattr(error, "status_code", 403), str(error))
508    if isinstance(error, AnthropicRateLimitError):
509        return APIStatusError(getattr(error, "status_code", 429), str(error))
510    if isinstance(error, AnthropicAPIConnectionError):
511        return APIConnectionError(str(error))
512    if isinstance(error, AnthropicAPITimeoutError):
513        return APITimeoutError(str(error))
514    return ChatProviderError(f"Anthropic error: {error}")

269        elif role == "tool":
270            if message.tool_call_id is None:
271                raise ChatProviderError("Tool message missing `tool_call_id`.")
272            if self._tool_message_conversion == "extract_text":
273                content = message.extract_text(sep="\n")
274            else:
275                content = message.content
276            block = _tool_result_message_to_block(message.tool_call_id, content)
277            return MessageParam(role="user", content=[block])
278
279        assert role in ("user", "assistant")
280        blocks: list[ContentBlockParam] = []
281        for part in message.content:
282            if isinstance(part, TextPart):
283                blocks.append(TextBlockParam(type="text", text=part.text))
284            elif isinstance(part, ImageURLPart):
285                blocks.append(_image_url_part_to_anthropic(part))
286            elif isinstance(part, ThinkPart):
287                if part.encrypted is None:
288                    # missing signature, strip this thinking block.
289                    continue
290                else:
291                    blocks.append(
292                        ThinkingBlockParam(
293                            type="thinking", thinking=part.think, signature=part.encrypted
294                        )
295                    )
296            else:
297                continue
298        for tool_call in message.tool_calls or []:
299            if tool_call.function.arguments:
300                try:
301                    parsed_arguments = json.loads(tool_call.function.arguments)
302                except json.JSONDecodeError as exc:  # pragma: no cover - defensive guard
303                    raise ChatProviderError("Tool call arguments must be valid JSON.") from exc
304                if not isinstance(parsed_arguments, dict):
305                    raise ChatProviderError("Tool call arguments must be a JSON object.")
306                tool_input = cast(dict[str, object], parsed_arguments)
307            else:
308                tool_input = {}
309            blocks.append(
310                ToolUseBlockParam(
311                    type="tool_use",
312                    id=tool_call.id,
313                    name=tool_call.function.name,
314                    input=tool_input,
315                )
316            )
317        return MessageParam(role=role, content=blocks)
